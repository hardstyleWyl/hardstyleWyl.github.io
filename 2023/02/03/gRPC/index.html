一、Protocol Buffers

2.数据类型

3.注意要点

1.字段规则 默认为单数字段

单数字段(Singular) ：这个字段只能出现0次或者1次

重复字段(Repeated)：如果你想做一个list或数组，可以使用重复字段的概念(声明前加 repeated),这个字段可以有任何数量的元素,并且值的顺序会被保留

2.字段的数值(Tag) 在C#中字段的名很重要 在protobuf中这个Tag值也很重

Tag值最小数值是1,最大为2^29 -1

注意:19000-19999之间的数不可以使用,这部分数值是保留的

从1-15的Tag数只占用一个字节所以他们用在频繁使用的字段上

16-2047占用两个字节所以他们用在不频繁的字段上

3.字段的默认值

4.保留的字段

5.为枚举值起别名

//option allow_alias = true; 起别名

6.其他要点

导入包 需要注意路径

同一proto可以定义多个消息类型,消息类型可以嵌套

外部访问内部通过(.)的方式

syntax = "proto3";//使用proto3语法
package my.project;//若不添加（option csharp_namespace）这个选项在C#中生成的代码Person类就在My.Project这个命名空间下
option csharp_namespace = "My.WebApis";//若设置了该选项则在C#中生成的代码就为My.WebApis的命名空间下了 但是他里面的包还为my.project
import "date.proto";//导入包
message Person{
	int32 id = 1;
    string name = 2;
    float height = 3;
    float weight = 4;
    bytes avatar = 5;//头像
    string email = 6;
    bool email_verifed = 7; //邮箱是否验证
    
    
    
	// repeated用来标明为重复字段 通常表示list
    repeated string phone_number = 8;//一个人可以有多个电话号码
	//保留Tag值
    reserved 9,11 to 20;//告诉其他开发人员Tag8，9，11-20不能用
    //保留的字段名
    reserved "foo","bar";//告诉其他开发人员 foo,bar字段名不能用
    
    
    
    //在消息内定义枚举
	enum Gender{
        option allow_alias = true; //允许为枚举使用别名
        NO_SPECIFIED = 0;//未定义性别 //枚举Tag值起始只能为0
       
        FEMALE = 1; //女性
        MALE = 2;   //男性
        
        WOMAN = 1; //女性别名
        MAN = 2;//男性别名
        //保留的Tag和枚举字段名
        reserved 7,8,9 to 12, 20 to 999;//保留枚举TAg值 max表明可能的最大值
        reserved "BOY", "GIRL";
    }
    
    
    Gender gender = 21;//在同一文件直接使用
    date birthdday = 22;//使用导入包的消息类型定义字段
    
    //可以在消息内定义消息类型
    message Address{
        string province = 1;//省份
        string city = 2;//
        string zip_code = 3;//邮编
        string street = 4;//街道
        string number = 5;//门牌号
    }

    repeated Address addresses = 23;//定义在消息内的消息类型可以直接用    --这个地址可能有多个地址定义成repeated(重复字段)类型
 
}

2.Startup中的配置

public class Startup
{
        public void ConfigureServices(IServiceCollection services)
        {
            //注册Grpc服务
            services.AddGrpc();
            services.AddSingleton<IClock, UtcClock>();
            services.AddSingleton<IDepartmentRepository, DepartmentRepository>();
            services.AddSingleton<IEmployeeRepository, EmployessRepository>();
        }

        public void Configure(IApplicationBuilder app, IWebHostEnvironment env)
        {
            if (env.IsDevelopment())
            {
                app.UseDeveloperExceptionPage();
            }
            app.UseRouting();
            app.UseEndpoints(endpoints =>
            {
                //在端点中间件注册grpc服务路由
                endpoints.MapGrpcService<DepartmentService>();
                endpoints.MapGrpcService<EmployessService>();
            });
        }
}

3.建立Api中的Repositories(存储库)

//部门存储库

public interface IDepartmentRepository
{
        Task<IEnumerable<Department>> GetAll();
        Task<Department> GetById(int id);
        Task<Department> Add(Department department);
}
public class DepartmentRepository : IDepartmentRepository
{
        private readonly List<Department> _departments = new List<Department>();
        public DepartmentRepository()
        {
            _departments.Add(new Department
            {
                Id = 1,
                EmployeeCount = 50,
                Name = "HR",
                Location = "Beijing"
            });
            _departments.Add(new Department
            {
                Id = 2,
                EmployeeCount = 40,
                Name = "SB",
                Location = "Beijing"
            });
            _departments.Add(new Department
            {
                Id = 3,
                EmployeeCount = 60,
                Name = "SD",
                Location = "Beijing"
            });
            _departments.Add(new Department
            {
                Id = 4,
                EmployeeCount = 66,
                Name = "SW",
                Location = "Beijing"
            });

        }
        public Task<Department> Add(Department department)
        {
            department.Id = _departments.Max(x => x.Id) + 1;
            _departments.Add(department);
            return Task.Run(() => department);
        }
        public Task<IEnumerable<Department>> GetAll()
        {
            return Task.Run(() => _departments.AsEnumerable());
        }
        public Task<Department> GetById(int id)
        {
            return Task.Run(() => _departments.FirstOrDefault(x => x.Id == id));
        }
}

4.建立proto 用于生成服务基类和实体

syntax = "proto3";

option csharp_namespace = "GRpcApi";
package API;

//服务形式为 rpc 服务方法(请求对象) returns (响应对象) {} //需要定义消息来进行请求与响应使用方法类似于Servlet
//消息可以表示的类型  请求/响应对象 实体对象

//定义Departments的服务和消息
service Departments{
	//rpc对应请求（Request）服务  returns对应响应（Response）服务
	rpc GetAll(GetAllDepartmentsRequest) returns (GetAllDepartmentsResponse){}
    rpc Add(AddDepartmentRequest) returns(AddDepartmentResponse){}
}

//GetAll请求服务定义  代表不用参数的请求
message GetAllDepartmentsRequest{}
//GetAll响应服务定义  代表响应值为一个Department类型的repeated(重复字段)列表
message GetAllDepartmentsResponse{
	repeated Department Departments = 1;
}
//实体对象消息
message Department {
    int32 id = 1;
    string name = 2;
    string location = 3;
    int32 employeeCount =4;
}
//AddDepartment请求服务定义
message AddDepartmentRequest{
    Department department = 1;
}
//AddDepartment响应服务定义
message AddDepartmentResponse{
    Department department = 1;
}

//----------------------------------Employee服务和消息定义 以下不做注释
service Employees{
    rpc GetByDepartmentId(GetByDepartmentIdRequest) returns (GetByDepartmentIdResponse){}
    rpc AddEmployee(AddEmployeeRequest) returns(AddEmployeeReqsponse){}
    rpc FireEmployee(FireEmployeeRequest) returns(FireEmployeeResponse){}
}

message GetByDepartmentIdRequest
{
    int32 departmentId = 1;
}

message GetByDepartmentIdResponse{
    repeated Employee employees = 1;
}

message Employee{
    int32 id = 1;
    int32 departmentId = 2;
    string fristName = 3;
    string lastName = 4;
    Gender gender = 5;
    bool fired = 6;
}
enum Gender{
    FEMALE = 0;
    MALE = 1;
}

message AddEmployeeRequest{
    Employee employee = 1;
}
message AddEmployeeReqsponse{
    Employee employee = 1;
}

message FireEmployeeRequest{
    int32 id = 1;
}
message FireEmployeeResponse{
    Employee employee = 1;
}

5.建立gRPC服务类 继承于刚才生成的服务基类

//这里的服务类似于Servlet

//部门gRPC服务类

namespace gRPCApi.GrpcServices
{

    //继承于Grpc.AspNetCore生成与之对应的类
    public class DepartmentService : Departments.DepartmentsBase 
    {
        //注入存储库服务
        private readonly IDepartmentRepository _departmentRepository;
        public DepartmentService(IDepartmentRepository departmentRepository)
        {
            _departmentRepository = departmentRepository;
        }
        
        //获取所有部门
        public override async Task<GetAllDepartmentsResponse> GetAll(GetAllDepartmentsRequest request, ServerCallContext context)
        {
            //这个方法的业务为    接收空的参数返回所有部门列表
            //创建响应对象
            var result = new GetAllDepartmentsResponse();
            //拿到部门列表
            var departments = await _departmentRepository.GetAll();
            //向响应对象的Departments属性列表中加入刚刚拿到的部门列表
            result.Departments.AddRange(departments);
            //返回响应
            return result;
        }

        //添加部门
        public override async Task<AddDepartmentResponse> Add(AddDepartmentRequest request, ServerCallContext context)
        {
            //这个方法的业务为    接收部门实体的参数返回部门实体
            var depatement = await _departmentRepository.Add(request.Department);
            var result = new AddDepartmentResponse
            {
                Department = depatement
            };
            return result;
        }
    }
}

//员工gRPC服务类

public class EmployessService : Employees.EmployeesBase
{
    	//注入部门存储库
        private readonly IEmployeeRepository _employeeRepository;
        public EmployessService(IEmployeeRepository employeeRepository)
        {
            _employeeRepository = employeeRepository;
        }
    
    	//添加员工
        public async override Task<AddEmployeeReqsponse> AddEmployee(AddEmployeeRequest request, ServerCallContext context)
        {
            //业务:从请求中获取员工加到存储库  并返回这个添加的员工
            var employee = await _employeeRepository.Add(request.Employee);
            return new AddEmployeeReqsponse
            {
                Employee = employee
            };
        }
    
    
        // 解雇员工
        public override async Task<FireEmployeeResponse> FireEmployee(FireEmployeeRequest request, ServerCallContext context)
        {
            //从请求中获取员工Id用在在存储库中进行解雇操作 并返回这个员工
            var employee = await _employeeRepository.Fire(request.Id);
            return new FireEmployeeResponse
            {
                Employee = employee
            };
        }
    
    
        // 根据部门ID查询这个部门的所有员工
        public override async Task<GetByDepartmentIdResponse> GetByDepartmentId(GetByDepartmentIdRequest request, ServerCallContext context)
        {
            //业务:从请求中获取部门id用于在存储库中查询这个部门中的所有的员工 并返回员工列表
            var employees = await _employeeRepository.GetByDepartmentId(request.DepartmentId);
            var result = new GetByDepartmentIdResponse();
            //添加到响应对象中
            result.Employees.AddRange(employees);
            return result;

        }
    }
